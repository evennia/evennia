    def test_cleanup_doesnt_delete_anything(self, mock_time):
        self.handler.add("a", 10)
        self.handler.add("b", 5)
        self.handler.add("c", 5)
        self.handler.add("d", 3.5)
        mock_time.return_value = 1.0
